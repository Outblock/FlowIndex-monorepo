# MCP Server Auth Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add API key authentication and rate limiting to the FlowIndex AI MCP server so external developers need a valid key to use it.

**Architecture:** Two-tier auth — admin key (env var, no limits) and developer keys (validated by proxying to Go backend's existing `X-API-Key` + `public.api_keys` table, rate limited at 60 req/min). Auth middleware wraps FastMCP's ASGI app.

**Tech Stack:** Python 3.12, FastMCP, Starlette (ASGI middleware), Go backend (existing key validation)

**Reference:** See `docs/plans/2026-03-07-mcp-auth-design.md` for design doc.

**Step 1: Add imports and rate limiter**

Add after existing imports (after `from db import ...` line):

```python
import time
import hashlib
from collections import defaultdict, deque

import config

# ---------------------------------------------------------------------------
# In-memory sliding-window rate limiter (per API key)
# ---------------------------------------------------------------------------
class RateLimiter:
    """Sliding window rate limiter. Tracks timestamps per key."""

    def __init__(self, max_requests: int = 60, window_seconds: int = 60):
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self._windows: dict[str, deque] = defaultdict(deque)

    def allow(self, key: str) -> bool:
        now = time.monotonic()
        window = self._windows[key]
        # Evict expired entries
        cutoff = now - self.window_seconds
        while window and window[0] < cutoff:
            window.popleft()
        if len(window) >= self.max_requests:
            return False
        window.append(now)
        return True


_rate_limiter = RateLimiter(
    max_requests=config.MCP_RATE_LIMIT,
    window_seconds=60,
)

# Cache validated developer keys for 5 min to avoid hitting Go backend on every request.
_key_cache: dict[str, float] = {}  # {key_hash: expiry_timestamp}
_KEY_CACHE_TTL = 300  # seconds
```

**Step 2: Add key validation function**

```python
async def _validate_developer_key(api_key: str) -> bool:
    """Validate a developer API key against the Go backend."""
    key_hash = hashlib.sha256(api_key.encode()).hexdigest()

    # Check cache first
    cached_expiry = _key_cache.get(key_hash)
    if cached_expiry and time.time() < cached_expiry:
        return True

    # Hit Go backend — send a lightweight request with X-API-Key header.
    # The backend's rate limit middleware validates the key and returns 401 if invalid.
    try:
        resp = httpx.get(
            f"{config.BACKEND_URL}/health",
            headers={"X-API-Key": api_key},
            timeout=5,
        )
        # 200 = key valid (or no key needed for /health, but the middleware still validates)
        # 401 = invalid key
        if resp.status_code == 401:
            return False
        # Cache valid keys
        _key_cache[key_hash] = time.time() + _KEY_CACHE_TTL
        return True
    except Exception:
        # If backend is unreachable, deny
        return False
```

**Step 3: Add ASGI auth middleware**

```python
from starlette.middleware import Middleware
from starlette.requests import Request
from starlette.responses import JSONResponse


async def auth_middleware(request: Request, call_next):
    """ASGI middleware that checks API key auth before forwarding to FastMCP."""

    # Skip auth if disabled
    if not config.MCP_AUTH_ENABLED:
        return await call_next(request)

    # Extract key from Authorization header or x-api-key
    auth_header = request.headers.get("authorization", "")
    api_key = ""
    if auth_header.lower().startswith("bearer "):
        api_key = auth_header[7:].strip()
    if not api_key:
        api_key = request.headers.get("x-api-key", "").strip()

    if not api_key:
        return JSONResponse(
            {"error": "unauthorized", "message": "API key required. Pass via Authorization: Bearer <key> header."},
            status_code=401,
        )

    # Admin key — allow with no rate limit
    if config.MCP_ADMIN_KEY and api_key == config.MCP_ADMIN_KEY:
        return await call_next(request)

    # Developer key — validate against Go backend
    if not await _validate_developer_key(api_key):
        return JSONResponse(
            {"error": "unauthorized", "message": "Invalid or inactive API key."},
            status_code=401,
        )

    # Rate limit check
    if not _rate_limiter.allow(api_key):
        return JSONResponse(
            {"error": "rate_limited", "message": f"Rate limit exceeded ({config.MCP_RATE_LIMIT} req/min)."},
            status_code=429,
        )

    return await call_next(request)
```

### Task 3: Verify Go Backend Key Validation Works for MCP

The Go backend's rate limit middleware at `ratelimit.go:200` checks `X-API-Key` header and calls `s.apiKeyResolver` which does:

```sql
UPDATE public.api_keys SET last_used = now()
WHERE key_hash = $1 AND is_active = true
RETURNING user_id
```

**Step 1: Verify the /health endpoint actually runs through rate limit middleware**

Check `server.go` — if `/health` is exempt from the rate limit middleware (line 192 of `ratelimit.go`), we need a different endpoint. If it IS exempt, the middleware won't validate the key at all.

Looking at `ratelimit.go:192`: `/health` IS exempt. So we need a different lightweight endpoint.

```go
// In server.go route registration:
v1.HandleFunc("/auth/verify-key", s.handleVerifyAPIKey).Methods("POST")

// In a new or existing handler file:
func (s *Server) handleVerifyAPIKey(w http.ResponseWriter, r *http.Request) {
	apiKey := strings.TrimSpace(r.Header.Get("X-API-Key"))
	if apiKey == "" {
		http.Error(w, `{"valid":false,"error":"missing X-API-Key header"}`, http.StatusUnauthorized)
		return
	}
	if s.apiKeyResolver == nil {
		http.Error(w, `{"valid":false,"error":"API key auth not configured"}`, http.StatusServiceUnavailable)
		return
	}
	hash := sha256Hash(apiKey)
	userID, err := s.apiKeyResolver(r.Context(), hash)
	if err != nil || userID == "" {
		w.Header().Set("Content-Type", "application/json")
		w.WriteHeader(http.StatusUnauthorized)
		w.Write([]byte(`{"valid":false}`))
		return
	}
	w.Header().Set("Content-Type", "application/json")
	w.Write([]byte(`{"valid":true}`))
}
```

**Step 3: Update MCP server's `_validate_developer_key` to use the new endpoint**

```python
async def _validate_developer_key(api_key: str) -> bool:
    key_hash = hashlib.sha256(api_key.encode()).hexdigest()

    cached_expiry = _key_cache.get(key_hash)
    if cached_expiry and time.time() < cached_expiry:
        return True

    try:
        resp = httpx.post(
            f"{config.BACKEND_URL}/flow/v1/auth/verify-key",
            headers={"X-API-Key": api_key},
            timeout=5,
        )
        if resp.status_code != 200:
            return False
        data = resp.json()
        if data.get("valid"):
            _key_cache[key_hash] = time.time() + _KEY_CACHE_TTL
            return True
        return False
    except Exception:
        return False
```
